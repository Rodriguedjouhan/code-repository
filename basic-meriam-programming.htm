function changeImage()

{


//alert("You clicked the image !");


var current_src = document.getElementById("image-rotator").getAttribute("src");

//alert(current_src);


if(current_src == "http://www.onclick.com/msis110/articles/on-the-charles.jpg")

{

current_src = "http://www.onclick.com/msis110/articles/community-boating-dock.jpg";

}

else

{

current_src = "http://www.onclick.com/msis110/articles/on-the-charles.jpg";

}


document.getElementById("image-rotator").setAttribute("src", current_src);


return true;

}


function changeImageSize()

{

// TODO get the width attribute value, if it is 475, set it to 100, else set it to 475


//alert("You want to change the image size !");


var current_width = document.getElementById("image-rotator").getAttribute("width");

//alert(current_width);


if(current_width == "475")

{

current_width = "100";

}

else

{

current_width = "475";

}


document.getElementById("image-rotator").setAttribute("width", current_width);


return true;

}
